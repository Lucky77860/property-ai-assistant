  <button onclick="sendMessage()">Send</button>
</div>

<script>
let step = 0;
let data = {};

const messages = document.getElementById("messages");

function addMessage(text, sender) {
  const div = document.createElement("div");
  div.className = sender;
  div.innerText = text;
  messages.appendChild(div);
  messages.scrollTop = messages.scrollHeight;
}

function botQuestion() {
  if(step === 0) addMessage("Assalamualaikum! Are you looking to Buy or Rent?", "bot");
  if(step === 1) addMessage("What is your budget?", "bot");
  if(step === 2) addMessage("Which area are you interested in?", "bot");
  if(step === 3) addMessage("What type of property? (House, Plot, Apartment)", "bot");
  if(step === 4) addMessage("Please enter your phone number.", "bot");
}

function sendMessage() {
  const input = document.getElementById("userInput");
  const value = input.value;
  if(value === "") return;

  addMessage(value, "user");

  if(step === 0) data.purpose = value;
  if(step === 1) data.budget = value;
  if(step === 2) data.area = value;
  if(step === 3) data.type = value;
  if(step === 4) {
    data.phone = value;

    let message = `New Property Lead:%0A` +
      `Purpose: ${data.purpose}%0A` +
      `Budget: ${data.budget}%0A` +
      `Area: ${data.area}%0A` +
      `Type: ${data.type}%0A` +
      `Phone: ${data.phone}`;

    addMessage("Thank you! Redirecting to WhatsApp...", "bot");

    setTimeout(() => {
      // Replace 923XXXXXXXXX with your WhatsApp number
      window.location.href = `https://wa.me/9299047194?text=${message}`;
    }, 2000);
  }

  step++;
  input.value = "";

  if(step < 5) botQuestion();
}

botQuestion();
</script>
